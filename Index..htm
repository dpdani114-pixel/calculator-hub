<script>

/* ========= LOAN ========= */

function calculateLoan(){

  let P = loanAmount.value;

  let r = interestRate.value/100/12;

  let n = loanYears.value*12;

  let emi = (P*r*Math.pow(1+r,n))/(Math.pow(1+r,n)-1);

  loanResult.innerText = "Monthly EMI: $" + emi.toFixed(2);

}

 

/* ========= BMI ========= */

function calculateBMI(){

  let bmi = weight.value / ((height.value/100)**2);

  let status = bmi<18.5?"Underweight":bmi<25?"Normal":bmi<30?"Overweight":"Obese";

  bmiResult.innerText = `BMI: ${bmi.toFixed(1)} (${status})`;

}

 

/* ========= AGE ========= */

function calculateAge(){

  let birth = new Date(dob.value);

  let today = new Date();

  let age = today.getFullYear() - birth.getFullYear();

  if (today < new Date(today.getFullYear(), birth.getMonth(), birth.getDate())) age--;

  ageResult.innerText = `Age: ${age} years`;

}

 

/* ========= DATE ========= */

function dateDiff(){

  let d1 = new Date(date1.value);

  let d2 = new Date(date2.value);

  let diff = Math.abs(d2-d1)/(1000*60*60*24);

  dateResult.innerText = `Difference: ${diff} days`;

}

 

/* ========= CURRENCY ========= */

const API_KEY = 1928be6b3338f085d0fd8542;

fetch(`https://v6.exchangerate-api.com/v6/${API_KEY}/codes`)

.then(r=>r.json())

.then(d=>{

  d.supported_codes.forEach(c=>{

    fromCur.innerHTML+=`<option>${c[0]}</option>`;

    toCur.innerHTML+=`<option>${c[0]}</option>`;

  });

});

 

function convertCurrency(){

  fetch(`https://v6.exchangerate-api.com/v6/${API_KEY}/latest/${fromCur.value}`)

  .then(r=>r.json())

  .then(d=>{

    let rate = d.conversion_rates[toCur.value];

    curResult.innerText = "Converted Amount: " + (curAmount.value*rate).toFixed(2);

  });

}

 

/* ========= TEMP ========= */

function convertTemp(){

  let v=tempValue.value;

  let r;

  if(tempFrom.value=="C") r=tempTo.value=="F"?v*9/5+32:tempTo.value=="K"?+v+273.15:v;

  if(tempFrom.value=="F") r=tempTo.value=="C"?(v-32)*5/9:tempTo.value=="K"?(v-32)*5/9+273.15:v;

  if(tempFrom.value=="K") r=tempTo.value=="C"?v-273.15:tempTo.value=="F"?(v-273.15)*9/5+32:v;

  tempResult.innerText = r.toFixed(2);

}

 

/* ========= LENGTH ========= */

const L={cm:1,m:100,inch:2.54,ft:30.48};

function convertLength(){

  let r = lenValue.value*L[lenFrom.value]/L[lenTo.value];

  lenResult.innerText = r.toFixed(2);

}
